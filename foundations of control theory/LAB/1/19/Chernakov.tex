	\subsection{Вариант - 19}	
	$
		x’ ’ +x’ =t, x(0)=x’ (0)=0 \\
		x\rightarrow X(p) \\
		x’\rightarrow pX(p) \\
		x’’\rightarrow p^{2}X(p) \\
		t\rightarrow\frac{1}{p^{2}} \\
		p^{2}X(p)+pX(p)=\frac{1}{p^{2}} \\
		p^{4}X(p)+p^{3}X(p)=1 \\
		X(p)=\frac{1}{p^{4}+p^{3}}=\frac{1}{p^{3}(p+1)} \\
		X(p)=\frac{p^{2}-p+1}{p^{3}}-\frac{1}{p+1}=1-t+\frac{t^{2}}{2}-exp(-t) \\
	$
